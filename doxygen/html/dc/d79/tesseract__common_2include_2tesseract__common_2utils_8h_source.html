56template <typename... Args>
57std::string strFormat(const std::string& format, Args... args)
58{
59 int size_s = std::snprintf(nullptr, 0, format.c_str(), args...) + 1; // Extra space for '\0'
60 if (size_s <= 0)
61 throw std::runtime_error("Error during formatting.");
62
63 auto size = static_cast<size_t>(size_s);
64 auto buf = std::make_unique<char[]>(size); // NOLINT
65 std::snprintf(buf.get(), size, format.c_str(), args...);
66 return std::string(buf.get(), buf.get() + size - 1); // We don't want the '\0' inside
67}

358template <typename FloatType>
359bool toNumeric(const std::string& s, FloatType& value)
360{
361 if (s.empty())
362 return false;
363
364 std::stringstream ss;
365 ss.imbue(std::locale::classic());
366
367 ss << s;
368
369 FloatType out;
370 ss >> out;
371
372 if (ss.fail() || !ss.eof())
373 return false;
374
375 value = out;
376 return true;
377}
